# Code changes for revised ERD March 2, 2024

DROP TABLE IF EXISTS School_Profile;
CREATE TABLE School_Profile (
    school_id int AUTO_INCREMENT PRIMARY KEY,
    school_name varchar(30),
    primary_location varchar(30),
    campus_number int,
    phone varchar(15),
    hiring boolean,
    website text,
    statement_of_faith text,
    accreditation varchar(20),
    teachers_employed varchar(20),
    student_enrollment varchar(20),
    email varchar(25),
    password varchar(40),
    photo longblob,
    logo longblob,
    last_accessed datetime
);

DROP TABLE IF EXISTS Teacher_Profile;
CREATE TABLE Teacher_Staff_Profile (
    teacher_staff_id INT AUTO_INCREMENT PRIMARY KEY,
    first_name varchar(15),
    last_name varchar(15),
    looking boolean,
    phone varchar(15),
    email varchar(25),
    password varchar(40),
    home_church varchar(30),
    resume longblob,
    testimony text,
    cover_letter longblob,
    headshot longblob,
    last_accessed datetime
);

DROP TABLE IF EXISTS Job_Posting;
CREATE TABLE Job_Posting (
    job_id int AUTO_INCREMENT PRIMARY KEY,
    school_id int,
    job_title varchar(50),
    job_description text,
    job_location varchar(50),
    interview_location varchar(50),
    contact_email varchar(50),
    salary_range text,
    preferred_degree varchar(50),
    required_degree varchar(50),
    preferred_experience text,
    required_experience text,
    posted_date datetime
);

DROP TABLE IF EXISTS Reference;
CREATE TABLE Reference (
    reference_id int AUTO_INCREMENT PRIMARY KEY,
    teacher_staff_id int,
    name varchar(30),
    relation varchar(30),
    relation_type varchar(15),
    phone varchar(15),
    email varchar(30)
);

DROP TABLE IF EXISTS Saved_Jobs;
CREATE TABLE Saved_Jobs (
    saved_id int PRIMARY KEY,
    job_id int,
    teacher_staff_id int
);

DROP TABLE IF EXISTS Job_Applications;
CREATE TABLE Job_Applications (
    saved_id int PRIMARY KEY,
    job_id int,
    teacher_staff_id int
);
